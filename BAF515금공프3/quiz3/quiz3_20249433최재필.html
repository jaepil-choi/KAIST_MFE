Grading Score

def grading(score):
    if not (0 <= score <= 100):
        return 'Score must be a number between 0 and 100!!'

    if 90 < score:
        grade = 'A'
    elif 80 < score:
        grade = 'B'
    elif 70 < score:
        grade = 'C'
    elif 60 < score:
        grade = 'D'
    else:
        grade = 'F'
    
    return f'Grade is {grade}!'
